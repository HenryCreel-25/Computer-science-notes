# 算法分析与设计

## 一、排序算法

### 1.选择排序

> 过程：**==先扫描数组，再找最小数，放到最前==**
>
> 两层循环：
> 
> > + 外层循环：确定当前待放置最小元素的位置
> > + 内层循环：在**未排序部分**（即下标 `i` 到 `n-1`）中寻找最小元素的位置
> 
> 代码中的min,i,j其实是作为“指针”的作用使用的，不能直接用arr[i]这类直接赋值这仨，否则后面较小的数将会直接覆盖前面的元素
> 
> ```cpp
> void Selection_sort(int arr[], int n)
> {
> 	for (int i = 0; i < n; i++)
> 	{
> 		int min = i;
> 		for (int j = i + 1; j < n; j++)
> 		{
> 			if (arr[j] < arr[min])
> 			{
> 				min = j;
> 			}
> 		}
> 		int temp = arr[i];
> 		arr[i] = arr[min];
> 		arr[min] = temp;
> 	}
> }
> ```
> 
>

### 2.冒泡排序

> 过程：==**从前两两对比，更大的往后移**==（从后比也行，更小的也行，往前移也行）
>
> 两层循环：
>
> > + 外层循环：控制排序的**轮数**（趟数）。每一轮将当前未排序部分的最大元素“冒泡”到正确位置（数组末尾）
> >
> > + 内层循环：在**当前未排序部分**（即从下标 `0` 到 `n-1-i`）中，通过相邻比较将较大的元素向后移动
>
> 解释：`for (int j = 0; j < n - 1 - i; j++)`
>
> > 因为大的数浮动到数组末尾，即浮动`i`次后，后`i`个数是有序的，前`n-i个`数无序，所以`j < n - 1 - i`是当前循环需要浮动的次数，而`j = 0`能让所有未排序的数进行排序.
>
> ```cpp
> void Bubble_sort(int arr[], int n)
> {
>     			//倒数第二个数排序完后，最后一个数就自动有序了
> 	for (int i = 0; i < n - 1; i++)		//外层循环
> 	{
> 		for (int j = 0; j < n - 1 - i; j++)	//内层循环
> 		{
> 			if (arr[j] > arr[j + 1])
> 			{
> 				int temp = arr[j];
> 				arr[j] = arr[j + 1];
> 				arr[j + 1] = temp;
> 			}
> 		}
> 	}
> }
> ```
>
> 



### 3.直接插入排序

> 过程：==**和前面的数比，找到对应位置插入**==
>
> 两层循环：
>
> > + 外层循环：遍历未排序部分的每个元素（从下标 1 开始，因为下标 0 默认已排序）
> > + 内层循环：在已排序部分（下标 0 到 i-1）中从后向前扫描。只要 `j >= 0` 且已排序元素 `arr[j]` 大于 `key`，就将该元素后移一位（`arr[j+1] = arr[j]`），然后 `j--` 继续向前比较
>
> ```cpp
> void Insertion_sort(int arr[], int n)
> {
> 	for (int i = 1; i < n; i++)		//外层循环
> 	{
> 		int key = arr[i];
> 		int j = i - 1;
> 
> 		for (j = i - 1; j >= 0 && arr[j] > key; j--)	//内层循环
> 		{
> 			arr[j + 1] = arr[j];//
> 		}
> 		arr[j + 1] = key;
> 	}
> }
> ```
>
> 



### 4.希尔排序

> 过程：==**对每一个子序列进行直接插入排序**==
>
> 通过gap（步长）来划分子序列
>
> 排序函数由三层for循环组成：
>
> > + **外层循环**：控制`gap`取值，初始为 `n/2`，每次循环后减半，直到 `gap` 为 0 结束
> > + **中层循环**：从 `gap` 开始遍历数组，保证**每个元素**都能在**其所属的子序列**中参与插入排序
> > + **内层循环**：对每个子序列进行插入排序。将当前元素 `arr[i]` 暂存，然后在同一个子序列中**向前比较**并后移元素（升序排法），直到找到正确位置插入
>
> ```cpp
> #include <iostream>
> 
> using namespace std;
> 			 //待排序数组   元素个数
> void Shell_sort(int arr[], int n)
> {
> 	int gap, i, j;
> 	for (gap = n / 2; gap > 0; gap /= 2)		//外层循环
> 	{
> 		for (i = gap; i < n; i++)				//中层循环
> 		{
> 			int temp = arr[i];//暂存当前arr[i]的值
> 					//确保j-gap不为负数&&前面的数大于后面的数
> 			for (j = i; j >= gap && arr[j - gap] > arr[j]; j -= gap)	//内层循环
> 			{
> 				arr[j] = arr[j - gap];//按步长将大数放在后面
> 			}
> 			arr[j] = temp;//无论是否进行了交换，都将arr[i]赋值给arr[j]
> 		}
> 	}
> }
> 
> void printArray(int arr[],int n)
> {
> 	for (int i = 0; i < n; i++)
> 	{
> 		cout << arr[i] << " ";
> 	}
> }
> 
> int main()
> {
> 	int arr[100];
> 	int n;
> 
> 	cin >> n;
> 
> 	for (int i = 0; i < n; i++)
> 	{
> 		cin >> arr[i];
> 	}
> 
> 	Shell_sort(arr, n);
> 	printArray(arr,n);
> 
> 	return 0;
> }
> ```
>
> 



### 5.快速排序

> 过程：**==以某个数为界，小于它的数放在其左边，大于的放在其右边==**
>
> ```cpp
> //数组arr是全局变量，这样不用作为函数参数，传递来传递去，特别是在递归中
> //l是待排序区间左端点，r是右端点
> void Quick_sort(int l,int r)
> {
> 	if (l >= r)//l==r说明待排序区间只有1个数，视为排序完成
> 	{
> 		return;
> 	}
>     //l+(r-l)/2能确保x的值在待排序数组里
> 	int x = arr[l + (r - l) / 2];
> 	Partition(l, r, x);
> 
>     //first和last是全局变量，便于在递归中不断更新自己的值
>     //为了防止底层的递归覆盖全局变量，用临时变量纪录first和last
> 	int left = first;
> 	int right = last;
>     //不断递归划分区间去排序
> 	Quick_sort(l,left-1);
> 	Quick_sort(right + 1, r);
> }
> 
> 	//l是待排序区间左端点，r是右端点，x是界
> void Partition(int l, int r, int x)
> {
> 	first = l;
> 	last = r;
> 	int i = l;
> 	/*while循环工作目的：
> 		从区间左端点处开始，i位置不断进行比较：
> 		1.i位置比x小的，arr[i]和arr[first]交换位置，i++,first++
> 		2.和x相等的，只用i++；
> 		3.比x大的，arr[i]和arr[last]交换，last--
> 		形象理解：i遍历区域的每一个数，比x小的数放在first左侧区域，比x大的数放在last左侧，相等的就不动
> 	*/
> 	while (i <= last)//当i==last时，说明last右侧已经都是i遍历过的数了，此时i和last同时指的该数为最后一个待比较数
> 	{
> 		if (arr[i] < x)
> 		{
> 			swap(arr[i], arr[first]);
> 			i++;
> 			first++;
> 		}
> 		else if (arr[i] == x)
> 		{
> 			i++;
> 		}
> 		else if (arr[i] > x)
> 		{
> 			swap(arr[i], arr[last]);
> 			last--;
> 		}
> 	}
> }
> ```

### 7.归并排序

> 过程：==**将整个数组不断划分为只有一个元素的最小子数组，再**==
>
> ```cpp
> static int arr[100];
> static int help[100];
> 
> 			//数组左端点了，右端点r
> void Merge_sort(int l, int r)
> {
>     //l == r时说明数组元素只有一个，天然有序
> 	if (l == r)
> 	{
> 		return;
> 	}
> 
>     //m为数组中点
> 	int m = (r + l) / 2;
>     //不断划分为子数组
> 	Merge_sort(l, m);
> 	Merge_sort(m + 1, r);
>     //在子数组中进行排序
> 	merge(l, m, r);
> }
> 
> 		//左      中        右
> void merge(int l, int m, int r)
> {
>     //a，b充当“指针”作用
> 	int a = l;
> 	int b = m + 1;
>     //i从第一个元素开始遍历
> 	int i = l;
> 
> 	while (a <= m && b <= r)
> 	{
>         //谁小就将谁拷贝到help数组
> 		if (arr[a] <= arr[b])
> 		{
> 			help[i] = arr[a];
> 			a++;
> 		}
> 		else
> 		{
> 			help[i] = arr[b];
> 			b++;
> 		}
> 		i++;
> 	}
> 	/*左侧指针和右侧指针必有一个越界，一个不越界
>     以下两个while其实只会生效一个，目的是为了将未越界指针所指的数拷贝到help
>     数组*/
> 	while (a <= m)
> 	{
> 		help[i++] = arr[a++];
> 	}
> 
> 	while (b <= r)
> 	{
> 		help[i++] = arr[b++];
> 	}
> 
> 	for (i = l; i <= r; i++)
> 	{
> 		arr[i] = help[i];
> 	}
> }
> ```
>

### 8.基数排序

> 基数排序是一种特殊排序，一是不涉及比较，二是只适用于数字排序
>
> 过程：==**从低位（个位）开始，将该数位出现过的每个数的次数记录下来，再按照出现次数进行排序**==
>
> e.g：5个数的个位分别是4，4，1，5，3，所以
>
> > + 1：1次
> > + 3：1次
> > + 4：2次
> > + 5：1次
>
> 更新为：
>
> > + 1：1个
> > + ≤3：2个
> > + ≤4：4个
> > + ≤5：5个
>
> 说明5应该放在arr[4]的位置，第一个4放在arr[3]，第二个4放在arr[2]，以此类推，对所有的数进行排序
>
> 多数位的数只是将上述过程在每一数位上重复执行

> ```cpp
> #include <iostream>
> 
> using namespace std;
> 
> int cnt[10];//记录数字0~9出现的次数
> int help[100];//临时数组储存排好序的数组
> 
> //基数排序
> void Radio_sort(int arr[], int n, int bits)
> {
>     //offset作用：配合模10来记录某数位的数字，每过一轮乘以10去获取更高位数位的数
>     //bits：最大数是几位数，决定了循环的次数
> 	for (int offset = 1; bits > 0; offset *= 10, bits--)
> 	{
>         //循环开始前将计数数组置零
> 		for (int i = 0; i < 10; i++)
> 		{
> 			cnt[i] = 0;
> 		}
> 		//1.记录该数位每个数字出现的次数
> 		for (int i = 0; i < n; i++)
> 		{
> 			int temp = (arr[i] / offset) % 10;
> 			cnt[temp]++;
> 		}
> 		//2.处理成前数累加的形式
> 		for (int i = 1; i < 10; i++)
> 		{
> 			cnt[i] += cnt[i - 1];
> 		}
> 		//3.从最后的数字开始排
> 		for (int i = n - 1; i >= 0; i--)
> 		{
> 			int temp = (arr[i] / offset) % 10;
> 			cnt[temp]--;
>             //临时数组
> 			help[cnt[temp]] = arr[i];
> 		}
>         
>         //4.临时数组中排好序的数组再赋值给原数组
> 		for (int i = 0; i < n; i++)
>    		{
> 			arr[i] = help[i];
> 		}
> 	}
> }
> 
> void printArr(int arr[], int n)
> {
> 	for (int i = 0; i < n; i++)
> 	{
> 		cout << arr[i] <<" ";
> 	}
> 	cout << "\n";
> }
> 
> int main()
> {
> 	int n;
> 	int arr[100];
> 
> 	cin >> n;
> 	for (int i = 0; i < n; i++)
> 	{
> 		cin >> arr[i];
> 	}
> 
> 	
> 	for (int i = 0; i < n; i++)
> 	{
> 		help[i] = 0;
> 	}
> 
> 	Radio_sort(arr, n, 2);
> 	printArr(arr, n);
> 
> 	system("pause");
> 	return 0;
> }
> ```
